<script src="https://cdn.jsdelivr.net/npm/marked/marked.min.js"></script>

<script>
    document.getElementById('fetch-doc-btn').addEventListener('click', async () => {
        const btn = document.getElementById('fetch-doc-btn');
        const loader = document.getElementById('loading-text');
        const container = document.getElementById('mdx-container');
        
        btn.disabled = true;
        loader.style.display = 'block';
        container.innerHTML = '';

        try {
            const response = await fetch('https://backend-alpha-01.vercel.app/api/get-content', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({
                    doc_id: "69d779ebc085c8093d5bcda3"
                })
            });

            const data = await response.json();

            if (response.ok && data.status === "success") {
                container.innerHTML = marked.parse(data.content);
                document.getElementById('fetch-doc-btn').style.display = 'none';
                
                if (typeof MathJax !== 'undefined' && MathJax.typesetPromise) {
                    MathJax.typesetPromise([container]).then(() => {
                        console.log("MathJax typesetting complete.");
                    }).catch(function (err) {
                        console.error('MathJax typesetting error: ', err.message);
                    });
                } else {
                    console.warn("MathJax is not loaded or typesetPromise is not available.");
                    setTimeout(() => {
                        if (typeof MathJax !== 'undefined') {
                            MathJax.typesetPromise([container]);
                        }
                    }, 1000);
                }
            } else {
                container.innerHTML = `<p style="color: red;">Error: ${data.error}</p>`;
            }
        } catch (error) {
            console.error("Error fetching document:", error);
            container.innerHTML = `<p style="color: red;">Error fetching document.</p>`;
        } finally {
            btn.disabled = false;
            loader.style.display = 'none';
        }
    });
</script>
